Chỉ kiểm tra trong chính element và các run trực tiếp của nó. """ # Chỉ tìm trong các run trực tiếp của element (không tìm quá xa) color_nodes = [] # 1. chỉ chấp nhận màu đỏ rõ ràng for c in unique_colors: val = c.get(f"{{{NS['w']}}}val") if val: val_lower = val.lower().strip() # Chỉ chấp nhận mã màu đỏ chuẩn (không dùng RGB detection lỏng) if val_lower in ['ff0000', 'c00000', 'e81123', 'a31515', 'dc143c', 'b22222', '8b0000']: return True # Chấp nhận tên màu "red" if val_lower == 'red': return True # Kiểm tra RGB nhưng với tiêu chuẩn chặt chẽ hơn if len(val_lower) == 6 and all(ch in '0123456789abcdef' for ch in val_lower): try: r_val = int(val_lower[0:2], 16) g_val = int(val_lower[2:4], 16) b_val = int(val_lower[4:6], 16) # Chỉ chấp nhận nếu đỏ rất rõ ràng: đỏ > 220 và xanh lá + xanh dương < 100 if r_val > 220 and (g_val + b_val) < 100: return True except: pass return False # ------------------------------ # 2️⃣ HÀM KIỂM TRA HIGHLIGHT ĐỎ # ------------------------------ def is_red_highlight(el): highlights = el.xpath('.//w:highlight', namespaces=NS) for h in highlights: val = h.get(f"{{{NS['w']}}}val") if val and val.lower() in HIGHLIGHT_REDS: return True return False # ------------------------------ # 3️⃣ HÀM KIỂM TRA GẠCH CHÂN # ------------------------------ def is_underlined(el): underlines = el.xpath('.//w:u', namespaces=NS) for u in underlines: val = u.get(f"{{{NS['w']}}}val") if val and val.lower() != "none": return True return False # ------------------------------ # 4️⃣ HÀM KIỂM TRA ĐÁP ÁN ĐÚNG # ------------------------------ def check_is_correct_answer(el): """ Kiểm tra xem element có phải là đáp án đúng không. Kiểm tra màu đỏ trong text (chỉ màu đỏ rõ ràng) if is_red_text(el): return True # 3. Kiểm tra gạch chân (chỉ nếu không có màu đỏ) if is_underlined(el): return True return False def extract_answer_char(text): match = re.match(r'^\s*#?([A-D])[\.\)]', text.strip()) return match.group(1) if match else None def is_fixed_option(text): return text.strip().startswith('#') def parse_group_tag(text): match = re.search(r'[#]?<(g[0-3])>', text) return match.group(1) if match else None # ------------------------------------------------- # 4️⃣.5️⃣ HÀM TẠO HEADER PARAGRAPH # ------------------------------------------------- def create_header_paragraph(text, is_bold=False): """Tạo một paragraph với text trong Word XML""" p = etree.Element(f"{{{NS['w']}}}p") r = etree.SubElement(p, f"{{{NS['w']}}}r") if is_bold: rpr = etree.SubElement(r, f"{{{NS['w']}}}rPr") etree.SubElement(rpr, f"{{{NS['w']}}}b") t = etree.SubElement(r, f"{{{NS['w']}}}t") t.text = text return p # ------------------------------------------------- # 5️⃣ ĐỌC BẢNG ĐÁP ÁN LOẠI 1 (2 dòng) # ------------------------------------------------- def parse_table_two_rows(tbl): answer_map = {} rows = tbl.xpath('.//w:tr', namespaces=NS) if len(rows) == 2: q_cells = rows[0].xpath('.//w:tc', namespaces=NS) a_cells = rows[1].xpath('.//w:tc', namespaces=NS) for i in range(1, min(len(q_cells), len(a_cells))): q_text = get_text_from_element(q_cells[i]) a_text = get_text_from_element(a_cells[i]) q = re.findall(r'\d+', q_text) a = re.findall(r'[A-D]', a_text.upper()) if q and a: answer_map[int(q[0])] = a[0] return answer_map # ------------------------------------------------- # 6️⃣ ĐỌC BẢNG ĐÁP ÁN LOẠI 2 (1 dòng) # ------------------------------------------------- def parse_table_one_row(tbl): answer_map = {} rows = tbl.xpath('.//w:tr', namespaces=NS) if len(rows) == 1: cells = rows[0].xpath('.//w:tc', namespaces=NS) for c in cells: t = get_text_from_element(c) pairs = re.findall(r'(\d+)\s*[\.\-]?\s*([A-D])', t.upper()) for q, a in pairs: answer_map[int(q)] = a return answer_map # ------------------------------------------------- # 7️⃣ ĐỌC DANH SÁCH LOẠI 3 (1.A 2.B ...) # ------------------------------------------------- def parse_answer_lines(body): answer_map = {} paras = body.xpath('.//w:p', namespaces=NS) for p in paras: t = get_text_from_element(p) pairs = re.findall(r'(\d+)\s*[\.\-]?\s*([A-D])', t.upper()) for q, a in pairs: answer_map[int(q)] = a return answer_map # ------------------------------------------------- # 8️⃣ HÀM ĐỌC BẢNG ĐÁP ÁN TỔNG HỢP # ------------------------------------------------- def read_answer_key(body): answer_map = {} tables = body.xpath('.//w:tbl', namespaces=NS) for tbl in tables: answer_map.update(parse_table_two_rows(tbl)) answer_map.update(parse_table_one_row(tbl)) # Dạng dòng rời answer_map.update(parse_answer_lines(body)) return answer_map # ------------------------------------------------- # 9️⃣ XỬ LÝ FILE DOCX + TRỘN ĐỀ # ------------------------------------------------- def create_header_paragraph(text, is_bold=False): """Tạo một paragraph với text trong Word XML""" p = etree.Element(f"{{{NS['w']}}}p") r = etree.SubElement(p, f"{{{NS['w']}}}r") rpr = etree.SubElement(r, f"{{{NS['w']}}}rPr") if is_bold: b = etree.SubElement(rpr, f"{{{NS['w']}}}b") t = etree.SubElement(r, f"{{{NS['w']}}}t") t.text = text return p def process_docx_shuffling(input_path, output_dir, num_variants=1, default_shuffle_mode='g3', school_name='', class_name='', subject_name='', exam_time=''): extract_dir = os.path.join(output_dir, "temp_extract") if os.path.exists(extract_dir): shutil.rmtree(extract_dir) with zipfile.ZipFile(input_path, 'r') as zip_ref: zip_ref.extractall(extract_dir) doc_xml_path = os.path.join(extract_dir, 'word', 'document.xml') xml_content = open(doc_xml_path, 'rb').read() tree = etree.fromstring(xml_content) body = tree.xpath('//w:body', namespaces=NS)[0] #
