import os import shutil import subprocess import time from fastapi import APIRouter, File, UploadFile from fastapi.responses import FileResponse router = APIRouter() # --- HÀM CHUYỂN ĐỔI DOCX -> PDF (Dùng LibreOffice) --- def convert_to_pdf(docx_path, output_dir): try: # LibreOffice convert command subprocess.run([ "libreoffice", "--headless", "--convert-to", "pdf", "--outdir", output_dir, docx_path ], check=True, stdout=subprocess.PIPE, stderr=subprocess.PIPE, timeout=60) filename = os.path.basename(docx_path) pdf_name = os.path.splitext(filename)[0] + ".pdf" return os.path.join(output_dir, pdf_name) except Exception as e: print(f"Lỗi LibreOffice: {e}") return None # --- API MỚI: CHỈ CHUYỂN ĐỔI, KHÔNG GỌI AI --- @router.post("/convert-only") async def convert_only(file: UploadFile = File(...)): print(f"--- Đang xử lý chuyển đổi file: {file.filename} ---") os.makedirs("uploads", exist_ok=True) temp_filename = f"uploads/{int(time.time())}_{file.filename}" try: # 1. Lưu file tạm with open(temp_filename, "wb") as buffer: shutil.copyfileobj(file.file, buffer) pdf_path = None # 2. Xử lý if file.filename.lower().endswith((".docx", ".doc")): # Nếu là Word -> Convert sang PDF pdf_path = convert_to_pdf(temp_filename, "uploads") elif file.filename.lower().endswith(".pdf"): # Nếu là PDF -> Dùng luôn pdf_path = temp_filename else: return {"error": "Chỉ hỗ trợ file .docx, .doc hoặc .pdf"} if not pdf_path or not os.path.exists(pdf_path): return {"error": "Lỗi Server: Không thể chuyển đổi file."} # 3. Trả về file PDF cho Client tự xử lý AI # Trình duyệt sẽ nhận file này và gọi Gemini API bằng IP của người dùng -> Không bị chặn return FileResponse(pdf_path, media_type='application/pdf', filename="converted_exam.pdf") except Exception as e: return {"error": f"Lỗi xử lý: {str(e)}"}
